document.getElementById('saveBtn').addEventListener('click', function() { const btn = this; // Ambil nilai open dan close const openVal = document.getElementById('openPrice').value; const closeVal = document.getElementById('closePrice').value; const pair = document.getElementById('mainPair').value; // Validasi input harga agar tidak kosong if (!openVal || !closeVal) { alert("Harap isi Price Open dan Price Close!"); return; } const openPrice = parseFloat(openVal); const closePrice = parseFloat(closeVal); // Perhitungan Pips Otomatis // Jika XAUUSD, selisih dikali 100. Selain itu dikali 1 (bisa kamu sesuaikan nanti) let pengali = (pair === "XAUUSD") ? 100 : 100; const hitungPips = ((closePrice - openPrice) * pengali).toFixed(2); const journalData = { pair: pair, h4_trend: document.querySelector('#H4 .trend').value, h4_rsi: document.querySelector('#H4 .rsi').value, h1_trend: document.querySelector('#H1 .trend').value, h1_rsi: document.querySelector('#H1 .rsi').value, m30_trend: document.querySelector('#M30 .trend').value, m30_rsi: document.querySelector('#M30 .rsi').value, m15_trend: document.querySelector('#M15 .trend').value, m15_rsi: document.querySelector('#M15 .rsi').value, candle: document.getElementById('mainCandle').value, open: openPrice, close: closePrice, notes: document.getElementById('mainNotes').value, pips: hitungPips // Data ini akan masuk ke kolom O }; btn.disabled = true; btn.innerText = 'Syncing...'; fetch(scriptURL, { method: 'POST', mode: 'no-cors', cache: 'no-cache', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(journalData) }) .then(() => { alert('Data Berhasil Dicatat ke Kolom O!'); btn.disabled = false; btn.innerText = 'Log Execution'; // Opsional: Kosongkan form setelah simpan document.getElementById('mainNotes').value = ""; }) .catch((error) => { console.error('Error:', error); alert('Gagal mengirim data!'); btn.disabled = false; btn.innerText = 'Log Execution'; }); });
